        private async Task<List<Capture>> ReadCapture(string[] lines)
        {
            try
            {
                if (this._error_off) throw new InvalidOperationException("Operation read \"Capture\" bot failed!");

                List<Capture> captures = new List<Capture>();
                int quantity = 0;
                foreach (string line in lines)
                {
                    quantity++;
                    if (quantity == 1) continue;
                    if (line == string.Empty) break;

                    string terms = line.Replace("\r", "");
                    string[] parts = terms.Split(';');
                    captures.Add(new Capture
                    {
                        environment = parts[0],
                        rotate = parts[1],
                        flash = parts[2],
                        result = (parts[3].ToUpper() == "SIM") || (parts[3].ToUpper() == "TRUE") ? true : false,
                    });
                }
                return captures;
            }
            catch (Exception ex)
            {
                this.error_message = ex.Message;
                this.OnError?.Invoke(this, this.error_message);
                return null;
            }
        }

        private string GetFilter(DecisionTree decision_tree, string node)
        {
            try
            {
                if (this._error_off) throw new InvalidOperationException("Operation get filter \"Capture\" bot failed!");

                if (node == ENVIRONMENT)
                {
                    if (decision_tree.children[(int)Environment.Light] == null) return LIGHT;
                    else if (decision_tree.children[(int)Environment.Dark] == null) return DARK;
                }
                if (node == ROTATE)
                {
                    if (decision_tree.children[(int)Rotate.Front] == null) return FRONT;
                    else if (decision_tree.children[(int)Rotate.Rear] == null) return REAR;

                if (node == FLASH)

                    if (decision_tree.children[(int)Flash.On] == null) return ON;
                    else if (decision_tree.children[(int)Flash.Off] == null) return OFF;
                    else if (decision_tree.children[(int)Flash.Auto] == null) return AUTO;

                return string.Empty;
            }
            catch (Exception ex)
            {
                this.error_message = ex.Message;
                this.OnError?.Invoke(this, this.error_message);
                return string.Empty;
            }

                DecisionTree decision = new DecisionTree();
                decision.level = level;
                decision.order = order;
                decision.node = value;
                decision.host = decision_tree;

                List<DecisionTree> childrens = new List<DecisionTree>();
                for (int quantity = 0; quantity < children; quantity++)
                {
                    DecisionTree tree = new DecisionTree();
                    tree = null;
                    childrens.Add(tree);
                }
                if (childrens.Count == 0) decision.children = null;
                else decision.children = childrens;

                //-----
                this._is_insert = true;
                this._order = order;

                return decision;
            }
            catch (Exception ex)
            {
                this.error_message = ex.Message;
                this.OnError?.Invoke(this, this.error_message);
                return null;
            }
        }

        public async Task<object> Load()
        {
            try
            {
                if (this._error_off) throw new InvalidOperationException("Operation load \"Capture\" bot failed!");

                DecisionTree decision_tree = null;
                string[] picker = await Picker();
                List<Capture> captures = new List<Capture>();
                captures = await ReadCapture(picker);
                int node = GetNode(captures);
                for (int quantity = 1; quantity <= node; quantity++)
                {
                    DecisionTree node_tree = new DecisionTree();
                    this._is_insert = false;
                    node_tree = Mount(captures, decision_tree, decision_tree);
                    if (node_tree != null) decision_tree = node_tree;
                }

                return decision_tree;
            }
            catch (Exception ex)
            {

                this.OnError?.Invoke(this, this.error_message);
                return null;
            }
        }
        
        
                private DecisionTree Mount(List<Capture> captures, DecisionTree decision_tree, DecisionTree decision_host)
	        {
	            try
	            {
	                if (this._error_off) throw new InvalidOperationException("Operation mount \"Capture\" bot failed!");
	
	                //-----
	                DecisionTree decision = new DecisionTree();
	                bool control_environment = false;
	                bool control_rotate = false;
	                bool control_flash = false;
	
	                if ((decision_host == null) && (decision_tree == null))
	                {
	                    string node = MountRoot(captures);
	                    int quantity  = GetList(captures, node);
	                    decision = Write(decision_host, 0, 1, node, quantity);
	                    return decision;
	                }
	                else if ((decision_host != null) && (decision_tree == null))
	                {
	                    //-----
	                    int order = this._order + 1;
	                    int level = this._level;
	
	                    decision = WriteNode(captures, decision_host, level, order);
	                    if (decision != null) return decision;
	                }
	
	                //-----
	                bool light = false;
	                bool dark = false;
	                bool front = false;
	                bool rear = false;
	                bool on = false;
	                bool off = false;
	                bool auto = false;
	
	                if (decision_tree.node.Contains(ENVIRONMENT))
	                {
	                    if (decision_tree.children[(int)Environment.Light] != null) 
	                        decision_tree.children[(int)Environment.Light] = Mount(captures, decision_tree.children[(int)Environment.Light], decision_tree);
	                    else light = true;
	                    if (decision_tree.children[(int)Environment.Dark] != null)
	                        decision_tree.children[(int)Environment.Dark] = Mount(captures, decision_tree.children[(int)Environment.Dark], decision_tree);
	                    else dark = true;
	                }
	                else control_environment = true;
	                if (decision_tree.node.Contains(ROTATE))
	                {
	                    if (decision_tree.children[(int)Rotate.Front] != null)
	                        decision_tree.children[(int)Rotate.Front] = Mount(captures, decision_tree.children[(int)Rotate.Front], decision_tree);
	                    else front = true;
	                    if (decision_tree.children[(int)Rotate.Rear] != null)
	                        decision_tree.children[(int)Rotate.Rear] = Mount(captures, decision_tree.children[(int)Rotate.Rear], decision_tree);
	                    else rear = false;
	                }
	                else control_rotate = true;
	                if (decision_tree.node.Contains(FLASH))
	                {
	                    if (decision_tree.children[(int)Flash.On] != null) 
	                        decision_tree.children[(int)Flash.On] = Mount(captures, decision_tree.children[(int)Flash.On], decision_tree);
	                    else on = true;
	                    if (decision_tree.children[(int)Flash.Off] != null)
	                        decision_tree.children[(int)Flash.Off] = Mount(captures, decision_tree.children[(int)Flash.Off], decision_tree);
	                    else off = true;
	                    if (decision_tree.children[(int)Flash.Auto] != null)
	                        decision_tree.children[(int)Flash.Auto] = Mount(captures, decision_tree.children[(int)Flash.Auto], decision_tree);
	                    else auto = true;
	                }
	                else control_flash = true;
	
	                //-----
	                decision = decision_tree;
	
	                //-----
	                if (!this._is_insert)
	                {
	                    if (!control_environment)
	                    {
	                        if (((light) && (dark)) || ((!light) && (dark)))
	                            return Rule(captures, decision_tree);
	                    }
	                    if (!control_rotate)
	                    {
	                        if (((front) && (rear)) || ((!front) && (rear)))
	                            return Rule(captures, decision_tree);
	                    }
	                    if (!control_flash)
	                    {
	                        if (((on) && (off) && (auto)) || 
	                            ((!on) && (off) && (off)) ||
	                            ((!on) && (!off) && (off)))
	                            return Rule(captures, decision_tree);
	                    }
	                }
	                return decision;
	            }
	            catch (Exception ex)
	            {
	                this.error_message = ex.Message;
	                this.OnError?.Invoke(this, this.error_message);
	                return null;
	            }
	        }
